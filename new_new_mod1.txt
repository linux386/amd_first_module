def depress(period='day', to_day=str_today):
    """Analyzes and records stock depression data."""
    path_depress = 'C:/Users/linux/OneDrive/stockdata/depress/depress_'
    from_day = (pd.to_datetime(to_day) - timedelta(days={'day': 60, 'week': 180, 'month': 730}[period])).strftime('%Y-%m-%d')
    all_stocks = select_stock('all', stock_last_day, stock_last_day)['Name'].to_list()
    depress_data = []

    for stock in all_stocks:
        stock_data = day_week_month_data(stock, from_day, to_day, period)
        stock_data['yesterday'] = stock_data.Close.shift(1)
        stock_data['minus'] = (stock_data['Close'] - stock_data['yesterday']) < 0
        consecutive_depress_days = stock_data['minus'].idxmax() if stock_data['minus'].any() else 0
        depress_data.append(consecutive_depress_days)

    depression_summary = pd.DataFrame({'Name': all_stocks, 'DepressionDays': depress_data}).sort_values(by='DepressionDays', ascending=False)
    depression_summary.to_excel(path_depress + period + str_today + '.xlsx')

def candle_graph(market='kospi', from_day='2020-01-01', period='week'):
    """Generates a candlestick graph for a specified market."""
    data_for_graph = day_week_month_data(market, from_day, period=period)[['Date', 'Open', 'High', 'Low', 'Close']]
    candlestick_trace = go.Candlestick(x=data_for_graph.Date, open=data_for_graph.Open, high=data_for_graph.High, low=data_for_graph.Low, close=data_for_graph.Close,
                                       increasing_line_color='red', decreasing_line_color='blue')
    fig = go.Figure(data=[candlestick_trace], layout=go.Layout(title=market))
    offline.iplot(fig, filename='candlestick')
	
def bokeh_chart(market='kospi', from_day='2019-01-01', period='month'):
    """Generates an interactive chart using Bokeh for stock data."""
    from bokeh.plotting import figure, show, output_notebook
    from math import pi
    output_notebook()
    market_data = day_week_month_data(market, from_day, period).set_index('Date')
    inc = market_data.Close > market_data.Open
    dec = market_data.Open > market_data.Close

    p = figure(x_axis_type='datetime', title=f"{market} Market Chart", tools='pan,wheel_zoom,box_zoom,reset,save')
    p.segment(market_data.index[inc], market_data.High[inc], market_data.index[inc], market_data.Low[inc], color='green')
    p.segment(market_data.index[dec], market_data.High[dec], market_data.index[dec], market_data.Low[dec], color='red')
    p.vbar(market_data.index[inc], 0.5, market_data.Open[inc], market_data.Close[inc], fill_color='green', line_color='green')
    p.vbar(market_data.index[dec], 0.5, market_data.Open[dec], market_data.Close[dec], fill_color='red', line_color='red')
    show(p)

def last_page(source):
    """Finds the last page number from a BeautifulSoup-parsed HTML source."""
    last_page_link = source.find('td', class_='pgRR').find('a')['href']
    last_page_number = int(last_page_link.split('page=')[-1])
    return last_page_number

def min_max(df, column):
    """Applies MinMax scaling to specified columns in a DataFrame."""
    df.columns = df.columns.str.lower()
    moving_averages = ma(df)

def ma(df):
    """Calculates moving averages for different periods and adds them to the DataFrame."""
    df.columns = df.columns.str.lower()
    df[['volume', 'close']] = df[['volume', 'close']].astype(float)
    for period in [5, 10, 15, 20, 30, 60, 120]:
        df[f'ma{period}'] = ta.MA(df, timeperiod=period)
    return df[['ma5', 'ma10', 'ma15', 'ma20', 'ma30', 'ma60', 'ma120']]

def close_ma(df, ma1='ma60', ma2='ma120'):
    """Plots moving averages based on the given DataFrame and selected MA periods."""
    df_with_ma = ma(df)
    scaled_data = MinMaxScaler().fit_transform(df_with_ma[[ma1, ma2]].values)
    scaled_df = pd.DataFrame(scaled_data, columns=[ma1, ma2])
    scaled_df.set_index(df['date'], inplace=True)
    scaled_df.plot(figsize=(16, 4))
    plt.title(df_with_ma['name'][0] if 'name' in df_with_ma.columns else 'Graph')
    plt.grid(True)
    plt.show()

def close_ma_vol(df, ma1='ma60', ma2='ma120', volume_col='volume'):

def make_dataset(stock_name, start_date):

def period_down(from_day, to_day):
    """Compares stock performance over a specified period."""
    start_time = time.time()
    query_min_close = "SELECT Name, MIN(Close) FROM market WHERE Date BETWEEN '{}' AND '{}' GROUP BY Name".format(from_day, to_day)
    min_close_df = pd.read_sql(query_min_close, engine)
    latest_stock_data = select_stock('all', 'latest')
    merged_df = pd.merge(min_close_df, latest_stock_data, on='Name')
    merged_df['diff'] = merged_df['Close'] / merged_df['MIN(Close)']
    merged_df.sort_values(by='diff', inplace=True)
    merged_df.to_excel(f'{path_down}period_down_{from_day}_to_{to_day}.xlsx')
    print(f'Processing completed in {time.time() - start_time:.2f} seconds')
