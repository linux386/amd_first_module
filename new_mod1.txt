def day_week_month_data(market='kospi', from_day='2020-01-01', to_day=str_today, period='month'):
    df = select_market_period(market, from_day) if market in ['kospi', 'kosdaq'] else select_stock(market, from_day, to_day)
    df['Date'] = pd.to_datetime(df['Date'])
    period_groups = df.groupby(pd.Grouper(key='Date', freq=period[0].upper()))
    aggregated_data = period_groups.agg({'Open': 'first', 'High': 'max', 'Low': 'min', 'Close': 'last', 'Volume': 'sum'})
    return aggregated_data.reset_index()

    # Further data transformation based on 'period'
    if period == 'day':
        return df  # This case is already handled in the previous refactoring
    else:
        # Reshaping array based on the length of the period
        arr = np.array([
            [p.iloc[-1]['Date'], p.iloc[0]['Open'], max(p['High']), min(p['Low']), p.iloc[-1]['Close'], sum(p['Volume'])]
            for p in period_groups])
        return pd.DataFrame(arr, columns=['Date', 'Open', 'High', 'Low', 'Close', 'Volume'])

def ma(df):
    df.columns = df.columns.str.lower()
    df[['volume', 'close']] = df[['volume', 'close']].astype(float)
    for period in [5, 10, 15, 20, 30, 60, 120]:
        df[f'ma{period}'] = ta.MA(df, timeperiod=period)
    return df[['ma5', 'ma10', 'ma15', 'ma20', 'ma30', 'ma60', 'ma120']]

def period_down(from_day, to_day):
    start_time = time.time()
    query_min_close = "SELECT Name, MIN(Close) FROM market WHERE Date BETWEEN '{}' AND '{}' GROUP BY Name".format(from_day, to_day)
    min_close_df = pd.read_sql(query_min_close, engine)
    latest_stock_data = select_stock('all', 'latest')
    merged_df = pd.merge(min_close_df, latest_stock_data, on='Name')
    merged_df['diff'] = merged_df['Close'] / merged_df['MIN(Close)']
    merged_df.sort_values(by='diff', inplace=True)
    merged_df.to_excel(f'{path_down}period_down_{from_day}_to_{to_day}.xlsx')
    print(f'Processing completed in {time.time() - start_time:.2f} seconds')

    def generate_quarterly_report(self, year, quarter):
        """ Generates a quarterly report for the specified year and quarter. """
        start_date, end_date = self.get_quarter_range(year, quarter)
        quarterly_data = self.fetch_data_range(start_date, end_date)
        self.save_report(quarterly_data, f'quarterly_report_{year}_Q{quarter}.xlsx')

    def get_quarter_range(self, year, quarter):
        """ Returns the start and end dates for the specified year and quarter. """
        if quarter == 1:
            start_month, end_month = 1, 3
        elif quarter == 2:
            start_month, end_month = 4, 6
        elif quarter == 3:
            start_month, end_month = 7, 9
        elif quarter == 4:
            start_month, end_month = 10, 12
        else:
            raise ValueError("Quarter must be between 1 and 4")

        start_date = datetime(year, start_month, 1)
        end_date = datetime(year, end_month, calendar.monthrange(year, end_month)[1])
        return start_date.strftime('%Y-%m-%d'), end_date.strftime('%Y-%m-%d')
