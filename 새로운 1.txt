def day_week_month_data(market='kospi', from_day='2020-01-01', to_day=str_today, period='month'):
    """
    주어진 시장의 데이터를 일/주/월 단위로 처리합니다.
    """
    if market=='kospi' or market=='kosdaq':
        df = select_market_period(market, from_day)
    else:
        df = select_stock(market,from_day,to_day)

	# 날짜 데이터 그룹화 및 처리
	df['Date'] = pd.to_datetime(df['Date'])
	months = [g for _, g in df.groupby(pd.Grouper(key='Date', freq='M'))]
	weeks = [g for _, g in df.groupby(pd.Grouper(key='Date', freq='W'))]
	columns = ['Date', 'Open', 'High', 'Low', 'Close', 'Volume']
	rows = []

	# 월별, 주별, 일별 데이터 처리
    if period == 'day':
        nick = 'day'
        df = df[['Date', 'Open', 'High', 'Low', 'Close', 'Volume']]
        df.columns = columns
        #df = df.set_index(df['date'])
        # return df
    if period == 'month':
        nick = 'month'
        period = months
    elif period == 'week':
        nick = 'week'
        period = weeks
		
	# 데이터 처리 루프
	for i in range(len(period)):
        try:
            rows.append(period[i].iloc[-1]['Date'])
            rows.append(period[i].iloc[0]["Open"])
            rows.append(max(period[i]['High']))
            rows.append(min(period[i]['Low']))
            rows.append(period[i].iloc[-1]['Close'])
            rows.append(sum(period[i]['Volume']))
    except Exception as e:
        print(f"Error processing data: {e}")

    arr = np.array(rows)

	# 데이터 반환 조건
	if nick == 'day':
		return df

	# 배열 재구성
	if nick == 'week' and len(period) > 177:
		arr1 = arr.reshape(len(period) - 1, 6)
	else:
		arr1 = arr.reshape(len(period), 6)

	# 데이터프레임 생성
	df = pd.DataFrame(data=arr1, columns=columns)

	# 최종 데이터프레임 반환
	return df

def depress(period='day', to_day=str_today):
    """
    특정 기간에 대한 데이터 처리를 수행하는 함수.
    """

    time_to_day = pd.to_datetime(to_day)

    if period == 'day':
        from_day = (time_to_day - timedelta(days=60)).strftime('%Y-%m-%d')
    elif period == 'week':
        from_day = (time_to_day - timedelta(days=180)).strftime('%Y-%m-%d')
    elif period == 'month':
        from_day = (time_to_day - timedelta(days=365*2)).strftime('%Y-%m-%d')
    else:
        print("depress( ['day', 'week', 'month'] )")
        return

    # 주식 이름 선택 및 리스트 변환
    df = select_stock('all', stock_last_day, stock_last_day)
    name = df['Name'].to_list()

    # 데이터 처리 루프
    count = 0
    depress_list = []
    for stock_name in name:
        df = day_week_month_data(market=stock_name, from_day=from_day, to_day=to_day, period=period)

	# 주식 데이터 계산 및 처리
	df['yesterday'] = df.Close.shift(1)
	df['minus'] = (df['Close'] - df['yesterday']) < 0
	df1 = df.sort_values(by=['Date'], axis=0, ascending=False, ignore_index=True)

	df3 = df2.sort_values(by=['count'], axis=0, ascending=False, ignore_index=True)
	
	# 조건에 따른 데이터 처리

def bokeh_chart(market='kospi', from_day='2019-01-01', period='month'):
    """
    주어진 시장과 기간에 대한 Bokeh 차트를 생성하는 함수.
    """
    from math import pi
    from bokeh.io import output_notebook, show
    from bokeh.plotting import figure
    from bokeh.layouts import gridplot
    output_notebook()

    # Bokeh 차트 데이터 준비
    df = day_week_month_data(market, from_day, period)
    df = df.set_index(df['Date'], drop=True)
    df.rename(columns={'Date': 'Date1'}, inplace=True)  # Bokeh 차트에서 Date 인덱스 사용을 위해
    mids = (df.Open + df.Close) / 2
    spans = abs(df.Close - df.Open)
    inc = df.Close >= df.Open
    dec = df.Open > df.Close

	# Bokeh 차트 설정 및 생성

	TOOLS = "pan,wheel_zoom,box_zoom,reset,save,crosshair"

	p_candlechart = figure(x_axis_type="datetime", tools=TOOLS, plot_width=900, plot_height=200, toolbar_location="left", title=market)
	p_candlechart.xaxis.major_label_orientation = pi/4
    p_candlechart.segment(df.index[inc], df.High[inc], df.index[inc], df.Low[inc], color="red")
    p_candlechart.segment(df.index[dec], df.High[dec], df.index[dec], df.Low[dec], color="blue")
    p_candlechart.vbar(df.index[inc], 0.5, df.Open[inc], df.Close[inc], fill_color="red", line_color="red",line_width=10)
    p_candlechart.vbar(df.index[dec], 0.5, df.Open[dec], df.Close[dec], fill_color="blue", line_color="blue",line_width=10)

	# 볼륨 차트 구성
	p_volumechart = figure(x_axis_type="datetime", tools=TOOLS, plot_width=900, plot_height=200, toolbar_location="left")
    p_volumechart.vbar(df.index, 0.5, df.Volume, fill_color="black", line_color="black",line_width=10)
	
	# 차트 표시
    p = figure(tools='crosshair', plot_width=900, toolbar_location="left")
    p = gridplot([[p_candlechart], [p_volumechart]], toolbar_location='left')	
	show(p)

def last_page(source):
    """
    웹 페이지의 마지막 페이지 번호를 찾는 함수.
    """
    last = source.find('td', class_='pgRR').find('a')['href']

def ma(DataFrame):
    """
    주어진 데이터프레임에 대해 여러 이동 평균을 계산하는 함수.
    """
    try:
        df = DataFrame.copy()
        df.columns = df.columns.str.lower()
        df[['volume', 'close']] = df[['volume', 'close']].astype(float)  # TA-Lib 계산을 위해 실수 타입으로 변환

        # 여러 이동 평균 계산
        df['ma5'] = ta.MA(df, timeperiod=5)
        df['ma10'] = ta.MA(df, timeperiod=10)
        df['ma15'] = ta.MA(df, timeperiod=15)
        df['ma20'] = ta.MA(df, timeperiod=20)
		df['ma30'] = ta.MA(df, timeperiod=30)
		df['ma60'] = ta.MA(df, timeperiod=60)
		df['ma120'] = ta.MA(df, timeperiod=120)		
		
    except Exception as e:
        print(f"Error in MA calculation: {e}")
    return df

def close_ma(df, select1='ma60', select2='ma120'):
    """
    주어진 데이터프레임에 대해 이동 평균 및 종가 데이터를 시각화하는 함수.
    """
    ma(df)
    source = MinMaxScaler()
    data = source.fit_transform(df[['close', select1, select2]].values)
    df1 = pd.DataFrame(data, columns=['close', select1, select2], index=df.index)
    df1.plot(figsize=(16, 4))
    plt.title(df['name'][0])
    plt.grid(True)
    plt.show()

def close_ma_vol(df, select1='ma60', select2='ma120', select3='volume'):
    """
    주어진 데이터프레임에 대해 이동 평균, 종가, 거래량 데이터를 시각화하는 함수.
    """
    try:
        ma(df)
        # 데이터 정규화 및 시각화
        source = MinMaxScaler()
        data = source.fit_transform(df[['close', select1, select2, select3]].values)
        df1 = pd.DataFrame(data, columns=['close', select1, select2, select3], index=df.index)
        df1.plot(figsize=(16, 4))
        plt.title(df['name'][0])
        plt.grid(True)
        plt.show()
    except Exception as e:
        print(f"Error in close_ma_vol: {e}")

def market_ma(df, select1, select2):

    시장 데이터에 대한 이동 평균을 계산하는 함수.
